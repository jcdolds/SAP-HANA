SELECT DISTINCT
  	   
	   TB1.COD_MATERIAL_R3,
	   TB1.COD_MATERIAL,
	   TB1.NOMBRE_MATERIAL,
	   TB1.MARCA,
	   TB1.COD_GRUPO_MATERIALES,
	   TB1.GRUPO_MATERIALES,
	   TB1.COD_SECTOR, 	
	   TB1.NOMBRE_SECTOR,
	   TB1.COD_GRUP_PRECIO_MATERIAL,
	   TB1.NOM_GRUP_PRECIO_MATERIAL,
	   TB1.JQ1,
	   TB2.VTEXT AS NOMBRE_JQ1,
	   TB1.JQ2,
	   TB3.VTEXT AS NOMBRE_JQ2,
	   TB1.JQ3,
	   TB4.VTEXT AS NOMBRE_JQ3,
	   TB1.JQ4,
	   TB5.VTEXT AS NOMBRE_JQ4,
	   TB1.JQ5,
	   TB6.VTEXT AS NOMBRE_JQ5,
	   TB1.VOLUMEN,
	   TB1.UNIDAD_VOLUMEN,
	   TB1.PRECIO_BASE,
	   TB1.UNIDAD_MEDIDA,
	   TB1.NUMERADOR,
	   TB1.DENOMINADOR,
	   TB1.COD_STATUS
	  
	   
	   FROM
(SELECT   LEFT(MARA.PRDHA,3) AS JQ1,
		LEFT(MARA.PRDHA,6) AS JQ2,
		LEFT(MARA.PRDHA,9) AS JQ3,
		LEFT(MARA.PRDHA,12) AS JQ4,
		LEFT(MARA.PRDHA,14) AS JQ5,
	    MARA.MATNR AS COD_MATERIAL,
		MARA.BISMT AS COD_MATERIAL_R3,
		MARA.MATNR AS COD_MATERIAL_HANA,
		MAKT.MAKTX AS NOMBRE_MATERIAL,
		T024X.LBTXT AS MARCA,
		MARA.MATKL AS COD_GRUPO_MATERIALES,
		T023T.WGBEZ AS GRUPO_MATERIALES,
		MARA.SPART AS COD_SECTOR,
		TSPAT.VTEXT AS NOMBRE_SECTOR,
		TBMKE.KONDM AS COD_GRUP_PRECIO_MATERIAL,
		T178T.VTEXT AS NOM_GRUP_PRECIO_MATERIAL,
		MARA.VOLUM AS VOLUMEN,
		MARA.VOLEH AS UNIDAD_VOLUMEN,
		KONP.KBETR AS PRECIO_BASE,
	   MARA.MEINS AS UNIDAD_MEDIDA,
	   MARM.UMREZ AS NUMERADOR,
	   MARM.UMREN AS DENOMINADOR,
	   MARA.MSTAE AS COD_STATUS
		
  FROM SAPABAP1.MARA
  LEFT JOIN SAPABAP1.TSPAT ON MARA.SPART = TSPAT.SPART  --- SECTOR ----
  				   AND TSPAT.MANDT = '300'
				   AND TSPAT.SPRAS = 'S'
  LEFT JOIN SAPABAP1.T023T ON MARA.MATKL = T023T.MATKL --- GRUPO ARTICULO ----
  LEFT JOIN SAPABAP1.T024X ON T024X.LABOR = MARA.LABOR ------- MARCAS --------
							AND T024X.MANDT = MARA.MANDT
   					        AND T024X.SPRAS = 'S'
  LEFT JOIN SAPABAP1.MAKT ON MARA.MATNR = MAKT.MATNR
  LEFT JOIN (SELECT DISTINCT MATNR, KONDM  FROM SAPABAP1.MVKE) TBMKE ON MARA.MATNR = TBMKE.MATNR
  LEFT JOIN SAPABAP1.T178T ON TBMKE.KONDM = T178T.KONDM
  							AND T178T.MANDT = '300'
  							AND T178T.SPRAS = 'S'
LEFT JOIN SAPABAP1.A600 ON A600.MATNR = MARA.MATNR
				        AND A600.DATBI LIKE '9999%'
LEFT JOIN SAPABAP1.KONP ON A600.KNUMH = KONP.KNUMH
LEFT JOIN SAPABAP1.MARM ON MARA.MEINS = MARM.MEINH
						 AND MARM.MATNR = MARA.MATNR
						 AND MARM.MANDT = '300'
  ) AS TB1
   					        
  					    
LEFT JOIN  (SELECT * FROM SAPABAP1.T179T WHERE MANDT = '300' AND SPRAS = 'S') TB2  ON TB2.PRODH = TB1.JQ1 
LEFT JOIN  (SELECT * FROM SAPABAP1.T179T WHERE MANDT = '300' AND SPRAS = 'S') TB3  ON TB3.PRODH = TB1.JQ2 
LEFT JOIN  (SELECT * FROM SAPABAP1.T179T WHERE MANDT = '300' AND SPRAS = 'S') TB4  ON TB4.PRODH = TB1.JQ3 
LEFT JOIN  (SELECT * FROM SAPABAP1.T179T WHERE MANDT = '300' AND SPRAS = 'S') TB5  ON TB5.PRODH = TB1.JQ4 
LEFT JOIN  (SELECT * FROM SAPABAP1.T179T WHERE MANDT = '300' AND SPRAS = 'S') TB6  ON TB6.PRODH = TB1.JQ5
