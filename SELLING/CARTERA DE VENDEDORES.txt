SELECT KNVP.KUNN2 AS COD_AGENTE,
	   KNA2.NAME1 AS NOMBRE_AGENTE,
	   KNVP.VKORG AS COD_ORG_VENTAS,
	   KNVP.VTWEG AS COD_CANAL,
	   TVTWT.VTEXT AS NOMBRE_CANAL,
	   KNVP.KUNNR AS COD_CLIENTE,
	   KNA1.NAME1 AS NOMBRE_CLIENTE,
	   KNVV.AUFSD AS BLOQUEADO
	   FROM SAPABAP1.KNVP
	   
INNER JOIN SAPABAP1.KNA1 ON KNA1.KUNNR = KNVP.KUNNR
INNER JOIN SAPABAP1.KNA1 KNA2 ON KNA2.KUNNR = KNVP.KUNN2
INNER JOIN SAPABAP1.TVTWT ON KNVP.VTWEG = TVTWT.VTWEG ----- NOMBRE CANAL ----
  				   AND TVTWT.MANDT = '300'
				   AND TVTWT.SPRAS = 'S'

LEFT JOIN SAPABAP1.KNVV ON KNVP.KUNNR = KNVV.KUNNR
					   AND KNVP.VKORG = KNVV.VKORG
					   AND KNVP.VTWEG = KNVV.VTWEG
